<Playground>
  <State initial={{disabled: false, value: value}}>
    {({ state, setState }) => {
      const onChange = (newValue)=>{
        console.log('mdx value >> ', newValue);
        setState({
          value:newValue
        })
      }
      return (
        LocaleProviderHoc(
          <AutoML.Derivative 
            data={data} 
            collapseData={collapseData}
            value={state.value} 
            onChange={onChange}
          >
          </AutoML.Derivative>
        )
      )
    }}
  </State>
</Playground>


<Playground>
  <State initial={{visible: false}}>
    {({ state, setState }) => {
      const Detail = AutoML.Derivative.Detail;
      const { visible } = state;
      console.log('detail >>> ', Detail, detailData)
      const handleCancel = ()=>{
        setState({
          visible:false
        })
      }
      const showModal = ()=>{
        setState({
          visible:true
        })
      }
      return (
        LocaleProviderHoc(
          <div>
            <p><a onClick={showModal}> 查看</a></p>
            <Detail visible={visible} data={detailData} width={800}	handleCancel={handleCancel}></Detail>
          </div>
        )
      )
    }}
  </State>
</Playground>
